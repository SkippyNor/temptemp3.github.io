const mergeImg = require('merge-img')
const jimp = require('jimp')

mergeImg([
  '1.png', 
  '2.png', 
  '3.png', 
  '4.png', 
], {
  direction: true,
})
.then((img) => {
  // Save image as file
  img.write('full.png', () => console.log('done'))
  img.resize(756, jimp.AUTO)
  img.write("small.png", () => console.log('image resize (done)'))
})
