mysql 1253 collation utf8 general not valid for character set utf8mb4

Documented work through collation utf8 general is not valid for character set utf8mb4 error while relocating database from one hosts to another. Document best read from bottom to top.

replace utf8mb4 with utf8

Approach

sed -i -e 's/\(CHARSET=utf8\)mb4\( COLLATE=utf8_general_ci\)/\1\2/

Check

grep -e 'CHARSET=utf8mb4 COLLATE=utf8_general_ci'

Sample


DROP TABLE IF EXISTS `wp_popularpostssummary`;
/*!40101 SET @saved_cs_client     = @@character_set_client */;
/*!40101 SET character_set_client = utf8 */;
CREATE TABLE `wp_popularpostssummary` (
  `ID` bigint(20) NOT NULL AUTO_INCREMENT,
  `postid` bigint(20) NOT NULL,
  `pageviews` bigint(20) NOT NULL DEFAULT '1',
  `view_date` date NOT NULL,
  `view_datetime` datetime NOT NULL,
  PRIMARY KEY (`ID`),
  KEY `postid` (`postid`),
  KEY `view_date` (`view_date`),
  KEY `view_datetime` (`view_datetime`)
) ENGINE=InnoDB AUTO_INCREMENT=7997 DEFAULT CHARSET=utf8mb4 COLLATE=utf8_general_ci;
/*!40101 SET character_set_client = @saved_cs_client */;

Error

ERROR 1253 (42000) at line 542: COLLATION 'utf8_general_ci' is not valid for CHARACTER SET 'utf8mb4'
